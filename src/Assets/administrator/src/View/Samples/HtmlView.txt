namespace {{namespace}}\Administrator\View\Samples;

\defined('_JEXEC') or die('Restricted Direct Access!');

use Joomla\CMS\Factory;
use Joomla\CMS\Language\Text;
use Joomla\CMS\MVC\View\GenericDataException;
use Joomla\CMS\MVC\View\HtmlView as BaseHtmlView;
use Joomla\CMS\Toolbar\Toolbar;
use Joomla\CMS\Toolbar\ToolbarFactoryInterface;
use Joomla\CMS\Toolbar\ToolbarHelper;
use {{namespace}}\Administrator\Helper\{{component_capitalize}}Helper;


class HtmlView extends BaseHtmlView
{
	/**
	 * An array of items
	 *
	 * @var  	array
	 *
	 * @since	{{version}}
	 */
	protected $items;

	/**
	 * The pagination object
	 *
	 * @var  	\JPagination
	 *
	 * @since	{{version}}
	 */
	protected $pagination;

	/**
	 * The model state
	 *
	 * @var  	\JObject
	 *
	 * @since	{{version}}
	 */
	protected $state;

	/**
	 * Form object for search filters
	 *
	 * @var  	\JForm
	 *
	 * @since	{{version}}
	 */
	public $filterForm;

	/**
	 * The active search filters
	 *
	 * @var  	array
	 *
	 * @since	{{version}}
	 */
	public $activeFilters;

	/**
	 * Display the view
	 *
	 * @param   string  $tpl  The name of the template file to parse; automatically searches through the template paths.
	 *
	 * @return  mixed  A string if successful, otherwise an Error object.
	 *
	 * @since	{{version}}
	 */
	public function display($tpl = null)
	{
		$this->items         = $this->get('Items');
		$this->pagination    = $this->get('Pagination');
		$this->state         = $this->get('State');
		$this->filterForm    = $this->get('FilterForm');
		$this->activeFilters = $this->get('ActiveFilters');

		$errors = $this->get('Errors');

		// Check for errors.
		if (count($errors = $this->get('Errors')))
		{
			throw new GenericDataException(implode("\n", $errors), 500);
		}

		$this->addToolbar();

		return parent::display($tpl);
	}

	/**
	 * Add the page title and toolbar.
	 *
	 * @return  void
	 *
	 * @since   {{version}}
	 */
	protected function addToolbar()
	{
		$canDo = {{component_capitalize}}Helper::getActions('{{prefix_component}}');
		$user  = Factory::getApplication()->getIdentity();

		// Get the toolbar object instance
		$toolbar = Toolbar::getInstance('toolbar');

		ToolbarHelper::title(Text::_('{{prefix_component_uppercase}}_TOOLBAR_LABEL'), 'book');

		if ($canDo->get('core.create'))
		{
			$toolbar->addNew('sample.add');
		}

		if ($canDo->get('core.edit.state'))
		{
			$dropdown = $toolbar->dropdownButton('status-group')
				->text('JTOOLBAR_CHANGE_STATUS')
				->toggleSplit(false)
				->icon('icon-ellipsis-h')
				->buttonClass('btn btn-action')
				->listCheck(true);

			$childBar = $dropdown->getChildToolbar();

			$childBar->publish('samples.publish')->listCheck(true);

			$childBar->unpublish('samples.unpublish')->listCheck(true);

			$childBar->archive('samples.archive')->listCheck(true);

			if ($user->authorise('core.admin'))
			{
				$childBar->checkin('samples.checkin')->listCheck(true);
			}

			if ($this->state->get('filter.published') !== -2)
			{
				$childBar->trash('samples.trash')->listCheck(true);
			}

			if ($this->state->get('filter.published') === -2 && $canDo->get('core.delete'))
			{
				$childBar->delete('samples.delete')
					->text('JTOOLBAR_EMPTY_TRASH')
					->message('JGLOBAL_CONFIRM_DELETE')
					->listCheck(true);
			}
		}

		if ($user->authorise('core.admin', '{{prefix_component}}') || $user->authorise('core.options', '{{prefix_component}}'))
		{
			$toolbar->preferences('{{prefix_component}}');
		}
	}
}
